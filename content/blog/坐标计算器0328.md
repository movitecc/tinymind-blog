        function gcd(a, b) {
            a = Math.abs(a);
            b = Math.abs(b);
            if (b > a) {
                [a, b] = [b, a];

                a = temp;
            }

        // äº¤æ±‡æ¨¡å¼è®¡ç®—
        function calculateIntersection() {
            hideError();
            
            // è·å–åˆå§‹å€¼å’Œæ­¥é•¿
            const initialX = parseFloat(document.getElementById('initialX').value);
            const stepX = parseFloat(document.getElementById('stepX').value);
            const initialY = parseFloat(document.getElementById('initialY').value);
            const stepY = parseFloat(document.getElementById('stepY').value);
            const maxValue = parseInt(document.getElementById('maxValue').value);
            
            // éªŒè¯æ‰€æœ‰è¾“å…¥çš„å€¼æ˜¯å¦ä¸ºæ•°å­—
            if (isNaN(initialX) || isNaN(stepX) || isNaN(initialY) || isNaN(stepY) || isNaN(maxValue)) {
                showError(t('invalidValues'));
                return;
            }
            
            // éªŒè¯maxValueæ˜¯åˆæ³•çš„æ•´æ•°
            if (maxValue <= 0 || !Number.isInteger(maxValue)) {
                showError(t('invalidMaxValue'));
                return;
            }
            
            // é˜²æ­¢maxValueè¿‡å¤§å¯¼è‡´è®¡ç®—æ—¶é—´è¿‡é•¿
            if (maxValue > 10000) {
                showError(t('maxValueTooBig'));
                return;
            }
            
            // æ£€æŸ¥ç‰¹æ®Šæƒ…å†µï¼šä¸¤ä¸ªæ­¥é•¿éƒ½ä¸º0
            if (stepX === 0 && stepY === 0) {
                showError(t('stepBothZero'));
                return;
            }
            
            // ç‰¹æ®Šæƒ…å†µ1ï¼šæ­¥é•¿ç›¸åŒ
            if (floatEquals(stepX, stepY) && Math.abs(stepX) > 0) {
                const diff = initialY - initialX;
                // æ£€æŸ¥å·®å€¼æ˜¯å¦æ˜¯æ­¥é•¿çš„æ•´æ•°å€
                if (floatEquals(diff % stepX, 0)) {
                    showError(t('sameSteps'));
                    return;
                }
            }
            
            // æ£€æŸ¥æ­¥é•¿çš„æœ€å¤§å…¬çº¦æ•°
            if (Math.abs(stepX) > 0 && Math.abs(stepY) > 0) {
                const gcdValue = gcd(stepX, stepY);
                if (gcdValue > 1 && !floatEquals(gcdValue, 1)) {
                    // åªæ˜¯è­¦å‘Šï¼Œä¸é˜»æ­¢è®¡ç®—
                    showSuccessMessage(t('gcdWarning'));
                }
            }
            
            // å¯»æ‰¾æ»¡è¶³æ¡ä»¶çš„n1å’Œn2å€¼
            const intersections = [];
            
            /*
             * è§£æ–¹ç¨‹ï¼ša + step_x * n1 = b + step_y * n2
             * é‡æ–°æ•´ç†ï¼šstep_x * n1 - step_y * n2 = b - a
             */
            const diff = initialY - initialX;
            
            // æ ¹æ®æ­¥é•¿çš„ä¸åŒæƒ…å†µåˆ†åˆ«å¤„ç†
            if (stepX === 0) {
                // å¦‚æœXæ­¥é•¿ä¸º0, Xå€¼æ’å®šä¸ºinitialX
                // æ£€æŸ¥æ˜¯å¦å­˜åœ¨n2ä½¿å¾— b + step_y * n2 = initialX
                if (stepY !== 0) {
                    const n2 = (initialX - initialY) / stepY;
                    if (Number.isInteger(n2) && n2 >= 1 && n2 <= maxValue) {
                        intersections.push({
                            n1: 1, // ä»»æ„å€¼ï¼Œè¿™é‡Œé€‰1
                            n2: n2,
                            value: initialX
                        });
                    }
                }
            } else if (stepY === 0) {
                // å¦‚æœYæ­¥é•¿ä¸º0, Yå€¼æ’å®šä¸ºinitialY
                // æ£€æŸ¥æ˜¯å¦å­˜åœ¨n1ä½¿å¾— a + step_x * n1 = initialY
                const n1 = (initialY - initialX) / stepX;
                if (Number.isInteger(n1) && n1 >= 1 && n1 <= maxValue) {
                    intersections.push({
                        n1: n1,
                        n2: 1, // ä»»æ„å€¼ï¼Œè¿™é‡Œé€‰1
                        value: initialY
                    });
                }
            } else {
                // ä¸¤ä¸ªæ­¥é•¿éƒ½ä¸ä¸º0çš„ä¸€èˆ¬æƒ…å†µ
                for (let n1 = 1; n1 <= maxValue; n1++) {
                    // è®¡ç®—n2: step_x * n1 - step_y * n2 = b - a
                    // å³ï¼šn2 = (step_x * n1 - (b - a)) / step_y
                    const n2 = (stepX * n1 - diff) / stepY;
                    
                    // n2å¿…é¡»æ˜¯æ•´æ•°ä¸”åœ¨èŒƒå›´å†…
                    if (Number.isInteger(n2) && n2 >= 1 && n2 <= maxValue) {
                        const intersectionValue = initialX + stepX * n1;
                        intersections.push({
                            n1: n1,
                            n2: n2,
                            value: intersectionValue
                        });
                    }
                }
            }
            
            // æŒ‰ç…§äº¤æ±‡å€¼å‡åºæ’åº
            intersections.sort((a, b) => a.value - b.value);
            
            // æ˜¾ç¤ºç»“æœ
            const summaryElem = document.getElementById('intersectionSummary');
            const tableBody = document.getElementById('intersectionTableBody');
            
            if (intersections.length === 0) {
                summaryElem.textContent = t('noIntersections');
                tableBody.innerHTML = '';
            } else {
                summaryElem.innerHTML = `
                    ${t('foundIntersections', intersections.length)}<br>
                    ${t('intersectionRange', maxValue)}
                `;
                
                // ç”Ÿæˆè¡¨æ ¼å†…å®¹
                let tableHTML = '';
                intersections.forEach(item => {
                    tableHTML += `
                        <tr>
                            <td>${item.n1}</td>
                            <td>${item.n2}</td>
                            <td>${item.value.toFixed(4)}</td>
                        </tr>
                    `;
                });
                
                tableBody.innerHTML = tableHTML;
            }
            
            // æ˜¾ç¤ºç»“æœåŒºåŸŸ
            document.getElementById('intersectionResults').style.display = 'block';
            document.getElementById('standardResults').style.display = 'none';
            
            // å¹³æ»‘æ»šåŠ¨åˆ°ç»“æœåŒºåŸŸ
            document.getElementById('intersectionResults').scrollIntoView({ behavior: 'smooth' });
        }
